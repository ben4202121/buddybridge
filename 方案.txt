```typescript
import { Notice } from 'obsidian';
import type { BuddyBridgeSettings, HistoryItem } from './types';

export class BuddyBridgeAPI {
    private requestId = 1;
    private gatewayUrl: string;
    private sessionId: string;
    private onHistoryAdd: ((item: HistoryItem) => void) | null = null;
    private abortController: AbortController | null = null;

    constructor(gatewayUrl: string, sessionId: string) {
        this.gatewayUrl = gatewayUrl;
        this.sessionId = sessionId;
    }

    setConfig(gatewayUrl: string, sessionId: string) {
        this.gatewayUrl = gatewayUrl;
        this.sessionId = sessionId;
    }

    setHistoryCallback(callback: (item: HistoryItem) => void) {
        this.onHistoryAdd = callback;
    }

    private generateId(): string {
        return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, (c) => {
            const r = Math.random() * 16 | 0;
            const v = c === 'x' ? r : (r & 0x3 | 0x8);
            return v.toString(16);
        });
    }

    getSessionId(): string {
        if (!this.sessionId) {
            this.sessionId = this.generateId();
        }
        return this.sessionId;
    }

    private createRequest(method: string, params: any) {
        return {
            jsonrpc: "2.0",
            id: this.requestId++,
            method: method,
            params: params
        };
    }

    private addHistory(type: string, query: string, result: string, status: 'success' | 'error' | 'pending' = 'success', filePath?: string) {
        const item: HistoryItem = {
            id: this.generateId(),
            type,
            query: query.substring(0, 200),
            result: result.substring(0, 500),
            time: new Date().toLocaleString(),
            filePath,
            status
        };
        this.onHistoryAdd?.(item);
    }

    async call(promptText: string, type: string = 'query', filePath?: string): Promise<string> {
        this.abortController?.abort();
        this.abortController = new AbortController();

        const sessionId = this.getSessionId();
        const requestBody = this.createRequest("session/prompt", {
            sessionId: sessionId,
            prompt: [{ type: "text", text: promptText }]
        });

        this.addHistory(type, promptText, '', 'pending', filePath);

        try {
            const response = await fetch(`${this.gatewayUrl}/api/v1/acp`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'x-codebuddy-request': '1',
                    'Accept': 'application/json, text/event-stream'
                },
                body: JSON.stringify(requestBody),
                signal: this.abortController.signal
            });

            if (!response.ok) {
                throw new Error(`Gateway 返回错误: ${response.status}`);
            }

            const reader = response.body?.getReader();
            if (!reader) {
                throw new Error('无法读取响应流');
            }

            let result = '';
            let buffer = '';
            const decoder = new TextDecoder();

            while (true) {
                const { done, value } = await reader.read();
                if (done) break;

                buffer += decoder.decode(value, { stream: true });
                const events = buffer.split('\n\n');
                buffer = events.pop() || '';

                for (const event of events) {
                    const parsed = this.parseSSEEvent(event);
                    if (parsed) {
                        const text = this.extractTextFromEvent(parsed);
                        if (text) result += text;
                    }
                }
            }

            if (buffer.trim()) {
                const lines = buffer.split('\n');
                let dataLine = '';
                for (const line of lines) {
                    if (line.startsWith('data:')) {
                        dataLine = line.substring(5).trim();
                    }
                }
                
                if (dataLine) {
                    try {
                        const parsed = JSON.parse(dataLine);
                        const text = this.extractTextFromEvent(parsed);
                        if (text) result += text;
                    } catch (e) {
                        console.warn('[BuddyBridge] SSE 数据可能不完整');
                    }
                }
            }

            this.addHistory(type, promptText, result, 'success', filePath);
            return result || '无返回内容';

        } catch (error) {
            if (error.name === 'AbortError') {
                throw new Error('请求已取消');
            }
            this.addHistory(type, promptText, error.message, 'error', filePath);
            throw error;
        }
    }

    private parseSSEEvent(eventStr: string): any | null {
        const lines = eventStr.split('\n');
        let dataLine = '';
        
        for (const line of lines) {
            if (line.startsWith('data:')) {
                dataLine = line.substring(5).trim();
            }
        }

        if (!dataLine) return null;

        try {
            return JSON.parse(dataLine);
        } catch (e) {
            return null;
        }
    }

    private extractTextFromEvent(data: any): string | null {
        if (data.method === 'session/update' && 
            data.params?.update?.sessionUpdate === 'agent_message_chunk' &&
            data.params.update.content?.type === 'text') {
            return data.params.update.content.text;
        }
        return null;
    }

    cancel() {
        this.abortController?.abort();
    }
}
```

```typescript
import { ItemView, WorkspaceLeaf, Notice, MarkdownView } from 'obsidian';
import type { HistoryItem } from '../types';

export const VIEW_TYPE_BUDDYBRIDGE = "buddybridge-panel";

export class BuddyBridgePanelView extends ItemView {
    private history: HistoryItem[] = [];
    private container: HTMLElement;
    private maxItems: number = 100;
    private filterType: string = 'all';

    constructor(leaf: WorkspaceLeaf, maxItems: number = 100) {
        super(leaf);
        this.maxItems = maxItems;
    }

    getViewType(): string {
        return VIEW_TYPE_BUDDYBRIDGE;
    }

    getDisplayText(): string {
        return "BuddyBridge 历史";
    }

    getIcon(): string {
        return "bot";
    }

    async onOpen() {
        this.container = this.contentEl.createDiv({ cls: 'buddybridge-panel' });
        this.render();
    }

    async onClose() {}

    setMaxItems(max: number) {
        this.maxItems = max;
        this.trimHistory();
        this.render();
    }

    addHistory(item: HistoryItem) {
        const pendingIndex = this.history.findIndex(h => 
            h.status === 'pending' && 
            h.type === item.type && 
            (h.query === item.query || h.query.startsWith(item.query.substring(0, 50)))
        );
        
        if (pendingIndex !== -1) {
            this.history[pendingIndex] = item;
        } else {
            this.history.unshift(item);
        }
        
        this.trimHistory();
        this.render();
    }

    private trimHistory() {
        if (this.history.length > this.maxItems) {
            this.history = this.history.slice(0, this.maxItems);
        }
    }

    clearHistory() {
        this.history = [];
        this.render();
    }

    exportHistory() {
        const markdown = this.history.map(h => `
## ${h.type} - ${h.time}

**状态**: ${h.status}  
**查询**: ${h.query}  
**文件**: ${h.filePath || '无'}  

\`\`\`
${h.result}
\`\`\`
`).join('\n---\n');

        const blob = new Blob([markdown], { type: 'text/markdown' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `buddybridge-history-${new Date().toISOString().slice(0,10)}.md`;
        a.click();
        URL.revokeObjectURL(url);
        new Notice('历史记录已导出');
    }

    private render() {
        this.container.empty();
        
        const header = this.container.createEl('div', { cls: 'buddybridge-panel-header' });
        header.createEl('h3', { text: 'BuddyBridge 历史', cls: 'buddybridge-panel-title' });
        
        const toolbar = header.createEl('div', { cls: 'buddybridge-toolbar' });
        
        const filterSelect = toolbar.createEl('select', { cls: 'buddybridge-filter' });
        const options = [
            { value: 'all', label: '全部' },
            { value: 'query', label: '查询' },
            { value: 'ingest', label: '导入' },
            { value: 'lint', label: 'Lint' },
            { value: 'drift-check', label: '漂移' },
            { value: 'batch-ingest', label: '批量' },
            { value: 'auto-lint', label: '自动' }
        ];
        for (const opt of options) {
            filterSelect.createEl('option', { value: opt.value, text: opt.label });
        }
        filterSelect.value = this.filterType;
        filterSelect.onchange = (e) => {
            this.filterType = (e.target as HTMLSelectElement).value;
            this.render();
        };

        const exportBtn = toolbar.createEl('button', { 
            text: '导出', 
            cls: 'buddybridge-btn buddybridge-btn-secondary' 
        });
        exportBtn.onclick = () => this.exportHistory();

        const clearBtn = toolbar.createEl('button', { 
            text: '清空', 
            cls: 'buddybridge-btn buddybridge-btn-danger' 
        });
        clearBtn.onclick = () => {
            this.clearHistory();
            new Notice('历史已清空');
        };

        const filtered = this.filterType === 'all' 
            ? this.history 
            : this.history.filter(h => h.type === this.filterType || h.type.startsWith(this.filterType));

        if (filtered.length === 0) {
            this.container.createEl('div', { 
                text: this.history.length === 0 
                    ? '暂无记录。使用右键菜单或命令开始查询。' 
                    : '当前筛选条件下无记录。',
                cls: 'buddybridge-empty' 
            });
            return;
        }

        const list = this.container.createEl('div', { cls: 'buddybridge-history-list' });
        
        for (const item of filtered) {
            const card = list.createEl('div', { 
                cls: `buddybridge-history-card buddybridge-status-${item.status}` 
            });
            
            const cardHeader = card.createEl('div', { cls: 'buddybridge-history-header' });
            cardHeader.createEl('span', { 
                text: item.type, 
                cls: `buddybridge-badge buddybridge-badge-${item.type.replace(/-/g, '')}` 
            });
            
            const meta = cardHeader.createEl('div', { cls: 'buddybridge-meta' });
            meta.createEl('span', { 
                text: item.time, 
                cls: 'buddybridge-time' 
            });
            if (item.status === 'pending') {
                meta.createEl('span', { text: '⏳', cls: 'buddybridge-status-icon' });
            } else if (item.status === 'error') {
                meta.createEl('span', { text: '❌', cls: 'buddybridge-status-icon' });
            }

            card.createEl('div', { text: item.query, cls: 'buddybridge-query' });

            if (item.filePath) {
                card.createEl('div', { 
                    text: `📄 ${item.filePath}`, 
                    cls: 'buddybridge-filepath' 
                });
            }

            if (item.result) {
                const resultDiv = card.createEl('div', { cls: 'buddybridge-result' });
                resultDiv.createEl('pre', { text: item.result });
            }

            const actions = card.createEl('div', { cls: 'buddybridge-actions' });
            
            if (item.result) {
                const copyBtn = actions.createEl('button', { 
                    text: '复制', 
                    cls: 'buddybridge-btn buddybridge-btn-small' 
                });
                copyBtn.onclick = () => {
                    navigator.clipboard.writeText(item.result);
                    new Notice('已复制到剪贴板');
                };
            }

            const insertBtn = actions.createEl('button', { 
                text: '插入文档', 
                cls: 'buddybridge-btn buddybridge-btn-small buddybridge-btn-primary' 
            });
            insertBtn.onclick = async () => {
                const activeView = this.app.workspace.getActiveViewOfType(MarkdownView);
                if (activeView) {

                    editor.replaceRange(text, cursor);
                    new Notice('已插入到文档');
                } else {
                    new Notice('请先打开一个 Markdown 文档');
                }
            };
        }
    }
}
```

```typescript
import { Notice, Editor, MarkdownView, TFile } from 'obsidian';
import { BuddyBridgeAPI } from '../api';
import type { BuddyBridgeSettings } from '../types';

export interface IBuddyBridgePlugin {
    api: BuddyBridgeAPI;
    settings: BuddyBridgeSettings;
    app: import('obsidian').App;
    panelView: import('../views/panel').BuddyBridgePanelView | null;
    saveSettings(): Promise<void>;
}

export abstract class BaseCommand {
    protected plugin: IBuddyBridgePlugin;

    constructor(plugin: IBuddyBridgePlugin) {
        this.plugin = plugin;
    }

    abstract get id(): string;
    abstract get name(): string;
    abstract get icon(): string;

    protected async execute(prompt: string, type: string, filePath?: string): Promise<string> {
        try {
            const result = await this.plugin.api.call(prompt, type, filePath);
            return result;
        } catch (error) {
            new Notice(`${this.name}失败: ${error.message}`);
            throw error;
        }
    }

    protected notify(message: string) {
        if (this.plugin.settings.showNotifications) {
            new Notice(message);
        }
    }
}
```

```typescript
import { Editor, MarkdownView, TFile, Notice } from 'obsidian';
import { BaseCommand, type IBuddyBridgePlugin } from './base';

export class IngestCommand extends BaseCommand {
    get id() { return 'buddybridge-ingest'; }
    get name() { return '导入知识库'; }
    get icon() { return 'book-plus'; }

    constructor(plugin: IBuddyBridgePlugin) {
        super(plugin);
    }

    async executeSelected(editor: Editor) {
        const selectedText = editor.getSelection();
        if (!selectedText) {
            new Notice('请先选中要导入的文本');
            return;
        }
        
        this.notify('正在导入知识库...');
        const result = await this.execute(`ingest: ${selectedText}`, 'ingest');
        this.notify(`导入完成: ${result.substring(0, 100)}...`);
        return result;
```

```typescript
import { TFile, Notice, TFolder } from 'obsidian';
import { BaseCommand, type IBuddyBridgePlugin } from './base';

export class BatchCommand extends BaseCommand {
    get id() { return 'buddybridge-batch-ingest'; }
    get name() { return '批量导入文件夹'; }
    get icon() { return 'folder-up'; }

    private concurrency = 3;

    constructor(plugin: IBuddyBridgePlugin) {
        super(plugin);
    }

    async executeFolderSimple(folderPath?: string) {
        const targetPath = folderPath || this.plugin.settings.batchIngestFolder;
        
        if (!targetPath) {
            new Notice('请先在设置中配置批量导入文件夹路径');
            return;
        }

        const folder = this.plugin.app.vault.getAbstractFileByPath(targetPath);
        if (!folder || !(folder instanceof TFolder)) {
            new Notice(`找不到文件夹: ${targetPath}`);
            return;
        }

        const files = this.getAllMarkdownFiles(folder);
        
        if (files.length === 0) {
            new Notice('指定文件夹中没有找到 Markdown 文件');
            return;
        }

        new Notice(`开始批量导入 ${files.length} 个文件（并发: ${this.concurrency}）...`);
        let success = 0;
        let failed = 0;

        for (let i = 0; i < files.length; i += this.concurrency) {
            const batch = files.slice(i, i + this.concurrency);
            const promises = batch.map(async (file) => {
                try {
                    const content = await this.plugin.app.vault.read(file);
                    await this.execute(`ingest: ${content}`, 'batch-ingest', file.path);
                    success++;
                    return { status: 'success' as const, file };
                } catch (error) {
                    failed++;
                    this.plugin.panelView?.addHistory({
                        id: '',
                        type: 'batch-ingest',
                        query: file.path,
                        result: `失败: ${error.message}`,
                        time: new Date().toLocaleString(),
                        filePath: file.path,
                        status: 'error'
                    });
                    return { status: 'error' as const, file, error };
                }
            });

            await Promise.all(promises);
            
            if (i + this.concurrency < files.length) {
                new Notice(`进度: ${Math.min(i + this.concurrency, files.length)}/${files.length}`);
            }
        }

        new Notice(`批量导入完成: ${success} 成功, ${failed} 失败`);
    }

    async executeCurrentFolder() {
        const file = this.plugin.app.workspace.getActiveFile();
        if (!file) {
            new Notice('请先打开一个文件');
            return;
        }
        
        const folder = file.parent;
        if (!folder) {
            new Notice('无法确定当前文件夹');
            return;
        }
        
        await this.executeFolderSimple(folder.path);
    }

    private getAllMarkdownFiles(folder: TFolder): TFile[] {
        const files: TFile[] = [];
        
        for (const child of folder.children) {
            if (child instanceof TFile && child.extension === 'md') {
                files.push(child);
            } else if (child instanceof TFolder) {
                files.push(...this.getAllMarkdownFiles(child));
            }
        }
        
        return files;
    }
}
```

```typescript
import { 
    Plugin, WorkspaceLeaf, TFile, debounce, 
    MarkdownView, Notice, Editor, TFolder
} from 'obsidian';
import { BuddyBridgeAPI } from './api';
import { BuddyBridgePanelView, VIEW_TYPE_BUDDYBRIDGE } from './views/panel';
import { DEFAULT_SETTINGS, type BuddyBridgeSettings, type HistoryItem } from './types';
import type { IBuddyBridgePlugin } from './commands/base';
import { IngestCommand } from './commands/ingest';
import { QueryCommand } from './commands/query';
import { LintCommand } from './commands/lint';
import { DriftCommand } from './commands/drift';
import { BatchCommand } from './commands/batch';
import { BuddyBridgeSettingTab } from './settings/tab';

export default class BuddyBridgePlugin extends Plugin implements IBuddyBridgePlugin {
    settings: BuddyBridgeSettings;
    api: BuddyBridgeAPI;
    panelView: BuddyBridgePanelView | null = null;
    
    private ingestCmd: IngestCommand;
    private queryCmd: QueryCommand;
    private lintCmd: LintCommand;
    private driftCmd: DriftCommand;
    private batchCmd: BatchCommand;
    
    private autoLintDebounced: ReturnType<<typeof debounce>;

    async onload() {
        await this.loadSettings();
        
        this.api = new BuddyBridgeAPI(this.settings.gatewayUrl, this.settings.sessionId);
        this.api.setHistoryCallback((item: HistoryItem) => {
            if (this.settings.autoSaveHistory) {
                this.panelView?.addHistory(item);
            }
        });

        this.ingestCmd = new IngestCommand(this);
        this.queryCmd = new QueryCommand(this);
        this.lintCmd = new LintCommand(this);
        this.driftCmd = new DriftCommand(this);
        this.batchCmd = new BatchCommand(this);

        this.autoLintDebounced = debounce(
            (file: TFile) => this.lintCmd.executeFile(file),
            this.settings.autoLintDelay,
            true
        );

        this.registerView(
            VIEW_TYPE_BUDDYBRIDGE,
            (leaf) => {
                this.panelView = new BuddyBridgePanelView(leaf, this.settings.maxHistoryItems);
                return this.panelView;
            }
        );

        this.addRibbonIcon('bot', 'BuddyBridge 历史面板', async () => {
            await this.activateView();
        });

        this.addRibbonIcon('search', 'BuddyBridge 快速查询', async () => {
            await this.queryCmd.executeQuick();
        });

        this.registerEvent(
            this.app.workspace.on('editor-menu', (menu, editor, view) => {
                menu.addItem((item) => {
                    item
                        .setTitle('BuddyBridge: 导入选中内容')
                        .setIcon('book-plus')
                        .onClick(async () => {
                            await this.ingestCmd.executeSelected(editor);
                        });
                });

                menu.addItem((item) => {
                    item
                        .setTitle('BuddyBridge: 查询选中内容')
                        .setIcon('search')
                        .onClick(async () => {
                            await this.queryCmd.executeSelected(editor, view);
                        });
                });

                menu.addItem((item) => {
                    item
                        .setTitle('BuddyBridge: Lint 检查')
                        .setIcon('check-circle')
                        .onClick(async () => {
                            await this.lintCmd.executeEditor(editor);
                        });
                });

                menu.addItem((item) => {
                    item
                        .setTitle('BuddyBridge: 导入当前文件')
                        .setIcon('files')
                        .onClick(async () => {
                            await this.ingestCmd.executeCurrentFile();
                        });
                });
            })
        );

        this.registerEvent(
            this.app.workspace.on('file-menu', (menu, file) => {
                if (file instanceof TFile && file.extension === 'md') {
                    menu.addItem((item) => {
                        item
                            .setTitle('BuddyBridge: 导入此文件')
                            .setIcon('book-plus')
                            .onClick(async () => {
                                await this.ingestCmd.executeFile(file);
                            });
                    });
                }
                
                if (file instanceof TFolder) {
                    menu.addItem((item) => {
                        item
                            .setTitle('BuddyBridge: 批量导入此文件夹')
                            .setIcon('folder-up')
                            .onClick(async () => {
                                await this.batchCmd.executeFolderSimple(file.path);
                            });
                    });
                }
            })
        );

        this.addCommand({
            id: 'buddybridge-drift-check',
            name: '漂移检查',
            editorCallback: async (editor: Editor, view: MarkdownView) => {
                await this.driftCmd.executeEditor(editor);
            }
        });

        this.addCommand({
            id: 'buddybridge-batch-ingest-folder',
            name: '批量导入默认文件夹',
            callback: async () => {
                await this.batchCmd.executeFolderSimple();
            }
        });

        this.addCommand({
            id: 'buddybridge-batch-ingest-current',
            name: '批量导入当前文件夹',
            callback: async () => {
                await this.batchCmd.executeCurrentFolder();
            }
        });

        this.addCommand({
            id: 'buddybridge-open-panel',
            name: '打开历史面板',
            callback: async () => {
                await this.activateView();
            }
        });

        this.addCommand({
            id: 'buddybridge-lint-current',
            name: 'Lint 当前文件',
            callback: async () => {
                await this.lintCmd.executeCurrentFile();
            }
        });

        this.registerEvent(
            this.app.vault.on('modify', (file) => {
                if (this.settings.autoLint && file instanceof TFile && file.extension === 'md') {
                    this.autoLintDebounced(file);
                }
            })
        );

        this.addSettingTab(new BuddyBridgeSettingTab(this.app, this));
    }

    onunload() {
        this.api.cancel();
    }

    async activateView() {
        const { workspace } = this.app;
        let leaf = workspace.getLeavesOfType(VIEW_TYPE_BUDDYBRIDGE)[0];
        
        if (!leaf) {
            const rightLeaf = workspace.getRightLeaf(false);
            if (rightLeaf) {
                await rightLeaf.setViewState({ type: VIEW_TYPE_BUDDYBRIDGE });
                leaf = rightLeaf;
            }
        }
        
        if (leaf) {
            workspace.revealLeaf(leaf);
        }
    }

    async loadSettings() {
        this.settings = Object.assign({}, DEFAULT_SETTINGS, await this.loadData());
    }

    async saveSettings() {
        await this.saveData(this.settings);
        this.api.setConfig(this.settings.gatewayUrl, this.settings.sessionId);
        
        this.autoLintDebounced = debounce(
            (file: TFile) => this.lintCmd.executeFile(file),
            this.settings.autoLintDelay,
            true
        );
    }
}
```

### 三、安装步骤

```bash
# 1. 进入 vault 插件目录
cd "D:\你的vault路径\.obsidian\plugins"

# 2. 创建项目目录
mkdir buddybridge
cd buddybridge

# 3. 初始化项目
npm init -y

# 4. 创建目录结构
mkdir src
mkdir src\views
mkdir src\commands
mkdir src\settings

# 5. 复制上面的所有文件到对应路径

# 6. 安装依赖
npm install obsidian @types/node esbuild builtin-modules tslib typescript

# 7. 编译
npx esbuild src/main.ts --bundle --outfile=main.js --format=cjs --target=es2018 --external:obsidian --external:electron --external:@codemirror/* --external:lezer --external:@lezer/* --external:codemirror --external:moment

# 8. 确认文件
dir main.js manifest.json styles.css

# 9. 重启 Obsidian
# 10. 设置 → 第三方插件 → 关闭安全模式 → 开启 BuddyBridge
```
